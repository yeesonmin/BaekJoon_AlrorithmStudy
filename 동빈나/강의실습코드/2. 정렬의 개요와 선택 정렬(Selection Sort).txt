#include<stdio.h>
#include <time.h>

/// <summary>
/// 선택정렬
/// </summary>
void sort(int* arr)
{
	int min = 9999, index = 0, temp = 0;
	for (int i = 0; i < 10; i++)
	{
		min = 9999;
		for (int j = i; j < 10; j++)
		{
			if (*(arr + j) <= min)
			{
				min = *(arr + j);
				index = j;
			}
		}
		temp = *(arr + i);
		*(arr + i) = min;
		*(arr + index) = temp;
	}
}

void main()
{
	clock_t start = clock(); // 시작 시간 저장
	int arr[10] = { 1, 10, 5, 8, 7, 6, 4, 3, 2, 9 };
	int* parr = arr;

	sort(parr);

	for (int i = 0; i < 10; i++)
	{
		printf("%d\n", *(parr + i));
	}
	clock_t end = clock(); // 코드가 끝난 시간 저장
	printf("\nTime: %lf\n", (double)(end - start) / CLOCKS_PER_SEC);

}
